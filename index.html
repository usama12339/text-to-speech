body { font-family: Arial, sans-serif; text-align: center; background: linear-gradient(to right, #4facfe, #00f2fe); color: white; padding: 20px; } .container { background: white; padding: 20px; border-radius: 10px; width: 50%; margin: auto; box-shadow: 0px 0px 10px rgba(0, 0, 0, 0.2); } h1 { color: #333; } textarea { width: 100%; height: 100px; border: 2px solid #4facfe; padding: 10px; } button { margin-top: 10px; padding: 10px; border: none; background: #4facfe; color: white; cursor: pointer; } button:hover { background: #00c6fb; } // Load available voices window.speechSynthesis.onvoiceschanged = function () { let voices = window.speechSynthesis.getVoices(); let voiceSelect = document.getElementById("voiceSelect"); voices.forEach((voice, index) => { let option = document.createElement("option"); option.text = voice.name; option.value = index; voiceSelect.appendChild(option); }); }; // Convert text to speech function speak() { let text = document.getElementById("text").value; let voiceIndex = document.getElementById("voiceSelect").value; let speech = new SpeechSynthesisUtterance(text); let voices = window.speechSynthesis.getVoices(); speech.voice = voices[voiceIndex]; window.speechSynthesis.speak(speech); } // Download the speech as an audio file function downloadAudio() { let text = document.getElementById("text").value; let speech = new SpeechSynthesisUtterance(text); let voices = window.speechSynthesis.getVoices(); speech.voice = voices[0]; // Default voice let audioURL = window.URL.createObjectURL(new Blob([text], { type: 'text/plain' })); let link = document.createElement("a"); link.href = audioURL; link.download = "speech.txt"; // Currently downloads as text (MP3 conversion needs server-side support) link.click(); }
